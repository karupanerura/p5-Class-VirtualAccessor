Class::VirtualAccessor - 

=head1 SYNOPSIS

  use Class::VirtualAccessor (
    rw => [qw/hoge fuga/],
    ro => [qw/foo/],
  );

  sub new {
      my $class = shift;
      my $args  = (@_ == 1) ? $_[0] : +{ @_ };

      bless(+{ %$args } => $class);
  }

  sub method {
      my $self = shift;

      $self->{hoge} = 'hogehoge';   # ok
      $self->{hoga} = 'hogehoge';   # typo! error!
  }
